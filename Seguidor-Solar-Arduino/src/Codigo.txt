#include <Servo.h>

// Definición de Servomotores [cite: 5, 6]
Servo servoHorizontal; 
Servo servoVertical;

// Pines de los Servos [cite: 6]
int pinServoH = 9;
int pinServoV = 10;

// Pines de las LDRs (Fotorresistencias) [cite: 6]
// Configuración: A0=Sup Izq, A1=Sup Der, A2=Inf Izq, A3=Inf Der
int ldrSI = A0; 
int ldrSD = A1; 
int ldrID = A2; 
int ldrII = A3;

// Variables de posición inicial
int posH = 90; 
int posV = 90;

// Ajustes de sensibilidad
int tolerancia = 10; // Rango de diferencia de luz ignorado para evitar vibraciones

void setup() {
  servoHorizontal.attach(pinServoH);
  servoVertical.attach(pinServoV);
  
  // Posicionar servos en el centro al iniciar
  servoHorizontal.write(posH);
  servoVertical.write(posV);
  delay(500);
}

void loop() {
  // Leer valores analógicos de las 4 LDRs [cite: 7]
  int vSI = analogRead(ldrSI);
  int vSD = analogRead(ldrSD);
  int vII = analogRead(ldrII);
  int vID = analogRead(ldrID);
  
  // Calcular promedios para determinar dirección 
  int promedioSup = (vSI + vSD) / 2; // Parte superior
  int promedioInf = (vII + vID) / 2; // Parte inferior
  int promedioIzq = (vSI + vII) / 2; // Parte izquierda
  int promedioDer = (vSD + vID) / 2; // Parte derecha

  // --- Lógica del Eje Vertical (Inclinación) ---
  if (abs(promedioSup - promedioInf) > tolerancia) {
    if (promedioSup > promedioInf && posV < 180) {
      posV++;
    } else if (promedioInf > promedioSup && posV > 0) {
      posV--;
    }
    servoVertical.write(posV);
  }

  // --- Lógica del Eje Horizontal (Rotación) ---
  if (abs(promedioIzq - promedioDer) > tolerancia) {
    if (promedioIzq > promedioDer && posH > 0) {
      posH--;
    } else if (promedioDer > promedioIzq && posH < 180) {
      posH++;
    }
    servoHorizontal.write(posH);
  }

  delay(50); // Pequeña pausa para suavizar el movimiento de los servos [cite: 2]
}
